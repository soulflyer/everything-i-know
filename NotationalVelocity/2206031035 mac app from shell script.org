#+TITLE: mac app from shell script
# Tags #mac #shell

From https://mathiasbynens.be/notes/shell-script-mac-apps

This works but there is no need to write the script, just use it as a guide to what to create.
#+begin_src shell
#!/usr/bin/env bash

APPNAME=${2:-$(basename "${1}" '.sh')};
DIR="${APPNAME}.app/Contents/MacOS";

if [ -a "${APPNAME}.app" ]; then
	echo "${PWD}/${APPNAME}.app already exists :(";
	exit 1;
fi;

mkdir -p "${DIR}";
cp "${1}" "${DIR}/${APPNAME}";
chmod +x "${DIR}/${APPNAME}";

echo "${PWD}/$APPNAME.app";
#+end_src
