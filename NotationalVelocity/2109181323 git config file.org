#+begin_src gitconfig
  [user]
          name = iain
          email = iain@soulflyer.co.uk
  [core]
          autocrlf = input
          excludesfile = ~/.gitignore
  [alias]
          hist = log --pretty=format:'%h %ad | %s%d [%an]' --graph --date=short
  [pull]
          rebase = false
  [init]
          defaultBranch = main
#+end_src
