#+TITLE: re-frame debugging events
# TAGS #clojure #debug

#+begin_src clojure
(rf/reg-event-db
  :say-hello
  (fn
    [db [_ who]]
    ;;(js/alert (str "hello " who))
    (println (str "hello " who))
    db))

(rf/reg-event-db
  :print-sub
  (fn  [db [_ sub]]
    (let [sub-val (rf/subscribe [sub])]
      (println (str "Sub: " @sub-val)))
    db))
#+end_src
