curl https://calendar.google.com/calendar/ical/iain.wood%40flexiana.com/private-7f0d1549d8783795c98292c8cb24c81e/basic.ics > /tmp/calendar
ical2orgpy /tmp/calendar Documents/org-mode/agenda/flexiana.org
