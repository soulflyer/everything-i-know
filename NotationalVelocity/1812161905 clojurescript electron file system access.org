Can write to the fie system from the electron render process (the js part)
#+begin_src clojurescript
(def fs (js/require "fs"))
(def path (js/require "path"))
(def current-dir (.resolve path "."))
(.writeFile fs (str current-dir "output-file") "whatever")
#+end_src
